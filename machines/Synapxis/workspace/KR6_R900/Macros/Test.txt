.PROGRAM Serveur

.DESCRIPTION

.INPUTS

.VARIABLES

    string message

.RETURN_TYPE none

.BODY


if(!tcpConnect(device))
    print("erreur connexion")
    return
end



while(true)
    if(tcpPopMessage(device, message))
        print("message: " + message)
        //traitement
        //renvoi des données
        if(!tcpSend(device, message))
            print(message)
            exit
        end
    end
end

if(tcpDisconnect(device))
    print("deconnexion réussie")
    return
else
    print("problème de déconnexion")
end
.END
